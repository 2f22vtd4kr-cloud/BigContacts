Core Concept
MCTS explores possible “outreach paths” in the entity graph (HNWI → Gatekeeper → Asset → Intermediary, etc.). It scores proximity (how “close to the body” a path gets) using heuristics based only on real registry data:
* Personal identifiers (direct phone/address in filings) = highest reward.
* Trusted gatekeeper with multiple shared assets = medium.
* Company/PR only = low.

1. mcts.py (new file in analysis module)
import random
import math
from collections import defaultdict
import networkx as nx
from typing import Dict, List, Tuple, Optional

class MCTSNode:
    def __init__(self, entity_id: str, path: List[str] = None):
        self.entity_id = entity_id
        self.path = path or [entity_id]
        self.visits = 0
        self.value = 0.0  # Cumulative proximity score
        self.children: Dict[str, MCTSNode] = {}

class ProximityMCTS:
    def __init__(self, graph: nx.DiGraph, max_depth: int = 4, c_param: float = 1.414):
        self.graph = graph  # NetworkX graph from real registry data
        self.max_depth = max_depth
        self.c_param = c_param  # Exploration constant

    def proximity_heuristic(self, current: str, target: str) -> float:
        """Score based ONLY on real data attributes (0.0 - 1.0)"""
        score = 0.0
        if self.graph.has_edge(current, target):
            edge_data = self.graph.edges[current, target]
            rel_type = edge_data.get('relationship_type', '')
            
            # Higher reward for closer/personal links from real data
            if rel_type in ['DIRECT_OWNER', 'PERSONAL_CONTACT']:
                score += 0.9
            elif rel_type in ['TRUSTED_GATEKEEPER', 'SHARED_ASSET']:
                score += 0.6
            elif rel_type in ['DIRECTOR_OF', 'BENEFICIAL_OWNER']:
                score += 0.4
            else:
                score += 0.2
        return min(1.0, score)

    def select(self, node: MCTSNode) -> MCTSNode:
        """UCT selection"""
        while node.children:
            node = max(node.children.values(), 
                      key=lambda n: (n.value / (n.visits + 1e-6)) + 
                                    self.c_param * math.sqrt(math.log(node.visits + 1) / (n.visits + 1e-6)))
        return node

    def expand(self, node: MCTSNode):
        """Add children from real graph neighbors"""
        if len(node.path) >= self.max_depth:
            return
        for neighbor in self.graph.successors(node.entity_id):
            if neighbor not in node.path:  # Avoid cycles
                child = MCTSNode(neighbor, node.path + [neighbor])
                node.children[neighbor] = child

    def simulate(self, node: MCTSNode, target_id: str) -> float:
        """Random rollout with heuristic scoring"""
        current = node.entity_id
        score = 0.0
        depth = len(node.path)
        
        for _ in range(self.max_depth - depth):
            neighbors = list(self.graph.successors(current))
            if not neighbors:
                break
            current = random.choice(neighbors)
            score += self.proximity_heuristic(current, target_id)
        
        return score / max(1, self.max_depth - depth)  # Average proximity

    def backpropagate(self, node: MCTSNode, reward: float):
        while node:
            node.visits += 1
            node.value += reward
            # Move to parent would require parent tracking; simplified here

    def search(self, start_id: str, target_id: str, iterations: int = 1000) -> List[Dict]:
        root = MCTSNode(start_id)
        for _ in range(iterations):
            leaf = self.select(root)
            self.expand(leaf)
            reward = self.simulate(leaf, target_id)
            self.backpropagate(leaf, reward)
        
        # Return top paths sorted by average value
        best_paths = []
        for child in root.children.values():
            if child.visits > 0:
                best_paths.append({
                    "path": child.path,
                    "proximity_score": child.value / child.visits,
                    "confidence": min(1.0, child.visits / 100)
                })
        return sorted(best_paths, key=lambda x: x["proximity_score"], reverse=True)[:5]
