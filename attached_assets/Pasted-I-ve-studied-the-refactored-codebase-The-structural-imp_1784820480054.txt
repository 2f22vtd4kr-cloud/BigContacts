import { logger } from "../logger";

export interface SocialDiscoveryResult {
  linkedinUrl: string | null;
  linkedinHeadline: string | null;
  twitterHandle: string | null;
  twitterBio: string | null;
  instagramHandle: string | null;
  personalWebsite: string | null;
  confidence: number;
  sources: string[];
}

export async function discoverSocialPresence(entity: {
  name: string;
  knownResidences?: string | null;
  metadata?: string | null;
}): Promise<SocialDiscoveryResult> {
  const result: SocialDiscoveryResult = {
    linkedinUrl: null, linkedinHeadline: null,
    twitterHandle: null, twitterBio: null,
    instagramHandle: null, personalWebsite: null,
    confidence: 0, sources: [],
  };

  // Step 1: Google Programmable Search for LinkedIn profile
  const googleQuery = `"${entity.name}" site:linkedin.com/in`;
  const googleResults = await googleSearch(googleQuery);
  if (googleResults.length > 0) {
    result.linkedinUrl = googleResults[0].url;
    result.linkedinHeadline = googleResults[0].snippet;
    result.confidence += 30;
    result.sources.push("google-linkedin");
  }

  // Step 2: DuckDuckGo for Twitter/X profile
  const ddgQuery = `"${entity.name}" site:twitter.com OR site:x.com`;
  const ddgResults = await duckDuckGoSearch(ddgQuery);
  if (ddgResults.length > 0) {
    const handleMatch = ddgResults[0].url.match(/(?:twitter\.com|x\.com)\/([^\/\?]+)/);
    if (handleMatch) {
      result.twitterHandle = handleMatch[1];
      result.twitterBio = ddgResults[0].snippet;
      result.confidence += 20;
      result.sources.push("ddg-twitter");
    }
  }

  // Step 3: Scrape LinkedIn profile page (public view)
  if (result.linkedinUrl) {
    const pageContent = await scrapePage(result.linkedinUrl);
    // Look for contact info section, personal website links
    const websiteMatch = pageContent.match(/href="(https?:\/\/[^"]*(?!linkedin\.com)[^"]+)"/);
    if (websiteMatch) {
      result.personalWebsite = websiteMatch[1];
      result.confidence += 15;
      result.sources.push("linkedin-website");
    }
  }

  // Step 4: Scrape Twitter bio for email/website
  if (result.twitterHandle) {
    const twitterPage = await scrapePage(`https://nitter.net/${result.twitterHandle}`);
    const emailMatch = twitterPage.match(/[\w.+-]+@[\w-]+\.[\w.-]+/);
    if (emailMatch) {
      result.sources.push("twitter-bio-email");
      result.confidence += 25;
    }
  }

  return result;
}

  // Step 1: Search for Telegram profile via t.me lookup
  // Generate username candidates from name
  const candidates = generateTelegramCandidates(entity.name);

    const response = await fetch(`https://t.me/${candidate}`);
    const html = await response.text();
    // t.me returns a page with the user's name and bio if the username exists

      result.telegramHandle = candidate;
      const bioMatch = html.match(/<meta name="description" content="([^"]+)"/);

      result.confidence += 40;
      break;

  // Use tgstat.com or similar public Telegram search

  return result;
}

function generateTelegramCandidates(name: string): string[] {
  const parts = name.toLowerCase().split(/\s+/);
  if (parts.length < 2) return [];
  const [first, last] = parts;
  return [
    `${first}${last}`,           // johnsmith
    `${first}_${last}`,          // john_smith
    `${first}.${last}`,          // john.smith
    `${first[0]}${last}`,        // jsmith
    `${first}_${last[0]}`,       // john_s
    `${last}`,                   // smith
    `${first}${last[0]}`,        // johns
  ];
}

export async function discoverViaFoundationFilings(entity: {
  name: string;
}): Promise<{ address?: string; email?: string; foundation?: string; confidence: number }> {
  // Step 1: Search ProPublica Nonprofit Explorer API
  const response = await fetch(
    `https://projects.propublica.org/nonprofits/api/v2/search.json?q=${encodeURIComponent(entity.name)}`
  );
  const data = await response.json();
  
  for (const org of data.organizations ?? []) {
    // Step 2: Get full filing (includes officer names + addresses)
    const filingResp = await fetch(
      `https://projects.propublica.org/nonprofits/api/v2/organizations/${org.ein}.json`
    );
    const filing = await filingResp.json();
    
    // Step 3: Check if entity is listed as officer/trustee
    const filingData = filing.filing_with_data;
    if (filingData?.officer_info) {
      for (const officer of filingData.officer_info) {
        if (nameMatch(officer.name, entity.name)) {
          return {
            address: officer.address,
            email: filingData.contact_email,
            foundation: org.name,
            confidence: 85, // IRS filing = very high confidence
          };
        }
      }
    }
  }
  return { confidence: 0 };
}
